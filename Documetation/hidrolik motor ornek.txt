public class HydraulicEngine : IDevice
{
    IProjeMixClient _client;
    int _address;
    public HydraulicEngine(IProjeMixClient client, int address)
    {
        _client = client;
        _address = address;
    }
    public string Name { get; set; }
    public bool Run { get { return _client.GetCoil(_address); } set { _client.SetCoil(_address, value); } }
}
