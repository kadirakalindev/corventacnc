 public class PistonRuler : IRuler
 {
     protected int _value;
     int _address;
     IProjeMixClient _client;
     public PistonRuler(IProjeMixClient client, int address)
     {
         _client = client;
         _address = address;
     }

     public string Name { get; set; }
     public int Value { get {_value= _client.GetRegister(_address); return _value; } }
     public string Position { get => registerTommCalculate().ToString()+" mm"; }
     public int RegisterCount { get; set; }
     public double Stroke_L { get; set; }
     private double registerTommCalculate()
     {
         double mm = Math.Round((double)Value * Stroke_L / (double)RegisterCount, 2);
         return mm;
     }
